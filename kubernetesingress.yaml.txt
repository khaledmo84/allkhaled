apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: alkhaled-bot
  namespace: default
  annotations:
    kubernetes.io/ingress.class: nginx
    cert-manager.io/cluster-issuer: letsencrypt-prod
spec:
  tls:
  - hosts:
    - alkhaled.example.com
    secretName: alkhaled-tls
  rules:
  - host: alkhaled.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: alkhaled-bot
            port:
              number: 8080
