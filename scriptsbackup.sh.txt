#!/bin/bash
BACKUP_DIR="./backups"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="$BACKUP_DIR/alkhaled_backup_$TIMESTAMP.tar.gz"

mkdir -p "$BACKUP_DIR"

tar -czf "$BACKUP_FILE" \
    alkhaled_exp.db \
    logs/ \
    .env \
    config.json \
    .rollback_snapshots/ \
    --exclude="*.pyc" 2>/dev/null || true

echo "Backup created: $BACKUP_FILE"
