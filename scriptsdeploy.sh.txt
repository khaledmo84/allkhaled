#!/bin/bash
set -e

echo "Building Docker image..."
docker build -t alkhaled:latest .

echo "Pushing to registry..."
docker tag alkhaled:latest ghcr.io/khaledmo84/alkhaled:latest
docker push ghcr.io/khaledmo84/alkhaled:latest

echo "Deploying to Fly.io..."
flyctl deploy

echo "Deployment complete."
