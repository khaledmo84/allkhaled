class PersistentMultiLevelCache:
    def __init__(self, cache_dir=".alkhaled_persistent"):
        os.makedirs(cache_dir, exist_ok=True)
        self._mem = {}
        self._ttl = {}
        self._sql = sqlite3.connect(os.path.join(cache_dir, "cache.db"), check_same_thread=False)
        self._sql.execute("PRAGMA journal_mode=WAL")
        self._sql.execute("CREATE TABLE IF NOT EXISTS cache (key TEXT PRIMARY KEY, val BLOB, exp REAL)")
        self._sql.commit()
        self._level = None
        try:
            import plyvel
            self._level = plyvel.DB(os.path.join(cache_dir, "leveldb"), create_if_missing=True)
        except:
            pass
        self._load_snapshot()
        self._running = True
        self._lock = asyncio.Lock()
        self._save_task = None

    def _load_snapshot(self):
        snap = os.path.join(os.path.dirname(self._sql.filename), "snapshot.json.gz")
        if os.path.exists(snap):
            try:
                with gzip.open(snap, 'rb') as f:
                    data = pickle.load(f)
                    self._mem = data.get('mem', {})
                    self._ttl = data.get('ttl', {})
                    now = time.time()
                    for k, exp in list(self._ttl.items()):
                        if exp < now:
                            self._mem.pop(k, None)
                            self._ttl.pop(k, None)
            except:
                pass

    async def start(self):
        self._save_task = asyncio.create_task(self._periodic_save())

    async def _periodic_save(self, interval=60):
        while self._running:
            await asyncio.sleep(interval)
            await self._save()

    async def _save(self):
        async with self._lock:
            data = {'mem': self._mem.copy(), 'ttl': self._ttl.copy()}
        try:
            with gzip.open(os.path.join(os.path.dirname(self._sql.filename), "snapshot.json.gz"), 'wb') as f:
                pickle.dump(data, f)
        except:
            pass

    async def get(self, key: str) -> Optional[Any]:
        if key in self._mem and time.time() < self._ttl.get(key, 0):
            return self._mem[key]
        cur = self._sql.execute("SELECT val, exp FROM cache WHERE key=?", (key,))
        row = cur.fetchone()
        if row and row[1] > time.time():
            val = pickle.loads(row[0])
            async with self._lock:
                self._mem[key] = val
                self._ttl[key] = row[1]
            return val
        if self._level:
            try:
                v = self._level.get(key.encode())
                if v:
                    d = pickle.loads(v)
                    if d.get('exp', 0) > time.time():
                        return d.get('val')
            except:
                pass
        return None

    async def set(self, key: str, val: Any, ttl: int = 60):
        exp = time.time() + ttl
        async with self._lock:
            self._mem[key] = val
            self._ttl[key] = exp
        self._sql.execute("INSERT OR REPLACE INTO cache (key, val, exp) VALUES (?, ?, ?)",
                          (key, pickle.dumps(val), exp))
        self._sql.commit()
        if self._level:
            try:
                self._level.put(key.encode(), pickle.dumps({'val': val, 'exp': exp}))
            except:
                pass

    async def close(self):
        self._running = False
        if self._save_task:
            self._save_task.cancel()
        self._sql.close()
        if self._level:
            self._level.close()

class RateLimiter:
    def __init__(self, max_req: int, period: int = 3600):
        self.max = max_req
        self.period = period
        self.count = 0
        self.reset = time.time() + period

    async def acquire(self) -> bool:
        now = time.time()
        if now > self.reset:
            self.count = 0
            self.reset = now + self.period
        if self.count >= self.max:
            return False
        self.count += 1
        return True

    valid_chains = ["ethereum", "bsc", "polygon", "arbitrum", "optimism", "base"]

class PriceSource:
    def __init__(self, name: str, url: str, query_tpl: Dict = None, limiter: RateLimiter = None):
        self.name = name
        self.url = url
        self.query_tpl = query_tpl or {}
        self.limiter = limiter
        self.latency = float('inf')
        self.ok = True

    async def fetch(self, session, token_in: str, token_out: str, chain: str) -> Optional[float]:
        if not self.ok or (self.limiter and not await self.limiter.acquire()):
            return None
        start = time.time()
        try:
            if self.name == "dexscreener":
                pair = f"{token_in}/{token_out}".upper()
                url = f"{self.url}/latest/dex/search?q={pair}"
                async with session.get(url, timeout=5) as r:
                    if r.status == 200:
                        data = await r.json()
                        for p in data.get('pairs', []):
                            if p.get('chainId', '').lower() == chain.lower() and p.get('baseToken', {}).get('symbol', '').upper() == token_in.upper():
                                price = float(p.get('priceUsd', 0))
                                if price > 0:
                                    self.latency = time.time() - start
                                    return price
            elif self.name == "coingecko":
                cid = "ethereum" if token_in.upper() == "ETH" else token_in.lower()
                url = f"{self.url}/api/v3/simple/price?ids={cid}&vs_currencies=usd"
                async with session.get(url, timeout=5) as r:
                    if r.status == 200:
                        data = await r.json()
                        price = float(data.get(cid, {}).get('usd', 0))
                        if price > 0:
                            self.latency = time.time() - start
                            return price
            elif self.name == "binance" and token_in.upper() == "ETH" and token_out.upper() == "USDT":
                url = f"{self.url}/api/v3/ticker/price?symbol=ETHUSDT"
                async with session.get(url, timeout=5) as r:
                    if r.status == 200:
                        data = await r.json()
                        price = float(data.get('price', 0))
                        if price > 0:
                            self.latency = time.time() - start
                            return price
            elif self.name in ["thegraph_uniswap_v3", "ormi", "goldsky", "envio"] and chain in self.query_tpl:
                addr_in = get_token_address(chain, token_in)
                addr_out = get_token_address(chain, token_out)
                if addr_in and addr_out:
                    query = f"{{ pools(where:{{token0:\"{addr_in}\",token1:\"{addr_out}\"}}) {{ token0Price }} }}"
                    async with session.post(self.query_tpl[chain], json={'query': query}, timeout=5) as r:
                        if r.status == 200:
                            data = await r.json()
                            pools = data.get('data', {}).get('pools', [])
                            if pools:
                                price = float(pools[0]['token0Price'])
                                if price > 0:
                                    self.latency = time.time() - start
                                    return price
        except:
            self.ok = False
        return None

class UltimatePriceFetcher:
    def __init__(self):
        self.cache = PersistentMultiLevelCache()
        self.sources = []
        self._init_sources()
        self.session = None
        self._round_robin_idx = 0
        self._monitor = APIMonitor()

    def _init_sources(self):
        self.sources.append(PriceSource("dexscreener", "https://api.dexscreener.com"))
        self.sources.append(PriceSource("coingecko", "https://api.coingecko.com", limiter=RateLimiter(50, 3600)))
        self.sources.append(PriceSource("binance", "https://api.binance.com"))
        tg = {"ethereum": "https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v3"}
        self.sources.append(PriceSource("thegraph_uniswap_v3", "", query_tpl=tg, limiter=RateLimiter(3000, 86400)))
        ormi_tpl = {"ethereum": "https://api.ormi.com/subgraphs/name/uniswap/uniswap-v3"}
        self.sources.append(PriceSource("ormi", "", query_tpl=ormi_tpl))

    async def start(self):
        self.session = aiohttp.ClientSession()
        await self.cache.start()
        asyncio.create_task(self._update_latencies())

    async def _update_latencies(self):
        while True:
            for src in self.sources:
                try:
                    start = time.time()
                    if src.name == "dexscreener":
                        await self.session.get("https://api.dexscreener.com/latest/dex/search?q=ETH", timeout=5)
                        src.latency = time.time() - start
                    elif src.name == "coingecko":
                        await self.session.get("https://api.coingecko.com/api/v3/ping", timeout=5)
                        src.latency = time.time() - start
                    elif src.name == "binance":
                        await self.session.get("https://api.binance.com/api/v3/ping", timeout=5)
                        src.latency = time.time() - start
                    elif src.name in ["thegraph_uniswap_v3", "ormi"]:
                        url = src.query_tpl.get("ethereum")
                        if url:
                            await self.session.post(url, json={'query': '{ _meta { block { number } } }'}, timeout=5)
                            src.latency = time.time() - start
                except:
                    src.latency = float('inf')
            await asyncio.sleep(300)

    async def get_price(self, token_in: str, token_out: str, chain: str = "ethereum") -> Optional[float]:
        valid, _ = validate_input(token_in, chain)
        if not valid:
            return None
        key = f"price:{token_in}:{token_out}:{chain}"
        cached = await self.cache.get(key)
        if cached:
            return cached
        sorted_src = sorted(self.sources, key=lambda s: s.latency)
        for src in sorted_src[:3]:
            price = await src.fetch(self.session, token_in, token_out, chain)
            if price and price > 0:
                await self.cache.set(key, price, ttl=30)
                return price
        return None

                change = random.gauss(trend, 0.02)
                base = base * (1 + change)
                data.append({'price': base, 'timestamp': time.time() - (days-i)*86400})

        return data

            await self.session.close()
        await self.cache.close()

        key = f"active_tokens_{chain}"
        cached = await self.cache.get(key)
        if cached:
            return cached
        tokens = set()
        # TheGraph method
        try:
            url = "https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v3"
            query = "{ tokens(first: 100, orderBy: totalValueLockedUSD, orderDirection: desc) { symbol } }"
            async with self.session.post(url, json={'query': query}, timeout=10) as r:
                if r.status == 200:
                    data = await r.json()
                    for t in data.get('data', {}).get('tokens', []):
                        tokens.add(t['symbol'])
        except:
            pass
        if not tokens:
            # Web3 fallback (example for Ethereum V2 factory)
            try:
                from web3 import Web3
                w3 = Web3(Web3.HTTPProvider("https://rpc.ankr.com/eth"))
                if w3.is_connected():
                    factory_addr = "0x5C69bEe701ef814a2B6a3EDD4B1652CB9cc5aA6f"

                    factory = w3.eth.contract(address=factory_addr, abi=abi)
                    length = await asyncio.to_thread(factory.functions.allPairsLength().call)
                    for i in range(min(length, 50)):
                        pair_addr = await asyncio.to_thread(factory.functions.allPairs(i).call)
                        tokens.add(pair_addr[:8])  # dummy, need token symbols
            except:
                pass
        if not tokens:
            tokens = {"WETH", "USDC", "DAI", "USDT"}
        result = list(tokens)[:50]
        await self.cache.set(key, result, ttl=3600)
        return result

    async def close(self):
        if self.session:
            await self.session.close()
        await self.cache.close()

    async def _worker(self):

class SecureSelfUpdater:
    def __init__(self, node_id, private_key_hex, code_path, ipfs=None):
        self.node_id = node_id
        self.private_key = private_key_hex
        self.code_path = code_path
        self.backup_dir = os.path.join(os.path.dirname(code_path), ".backups")
        os.makedirs(self.backup_dir, exist_ok=True)
        self.ipfs = ipfs
        self._rollback_task = None

    def _hash_file(self, path):
        h = hashlib.sha256()
        with open(path, 'rb') as f:
            for chunk in iter(lambda: f.read(65536), b''):
                h.update(chunk)
        return h.hexdigest()

    def _sign(self, data):
        import coincurve
        pk = coincurve.PrivateKey.from_hex(self.private_key)
        return pk.sign(data.encode()).hex()

    def _verify(self, data, sig, pubkey):
        import coincurve
        pub = coincurve.PublicKey.from_hex(pubkey)
        return pub.verify(bytes.fromhex(sig), data.encode())

    async def fetch_and_verify(self, cid, signature, pubkey, peer_hash):
        if not self._verify(cid, signature, pubkey):
            return None
        temp = tempfile.mktemp(suffix='.py')
        if self.ipfs and hasattr(self.ipfs, 'fetch'):
            success = self.ipfs.fetch(cid, temp)
            if not success:
                return None
        else:
            # simulate
            with open(temp, 'w') as f:
                f.write("# dummy")
        new_hash = self._hash_file(temp)
        if new_hash != peer_hash:
            return None
        backup = os.path.join(self.backup_dir, f"backup_{int(time.time())}.py")
        shutil.copy2(self.code_path, backup)
        shutil.move(temp, self.code_path)
        self._start_rollback_monitor(backup)
        return backup

    def _start_rollback_monitor(self, backup_path, timeout=300):
        async def monitor():
            await asyncio.sleep(timeout)
            if not await self._is_system_healthy():
                logger.warning("System unhealthy after update, rolling back...")
                shutil.copy2(backup_path, self.code_path)
                logger.info("Rollback complete")
        self._rollback_task = asyncio.create_task(monitor())

    async def _is_system_healthy(self):
        # Check health endpoint (default port 8080)
        try:
            async with aiohttp.ClientSession() as session:
                async with session.get("http://localhost:8080/health", timeout=5) as r:
                    return r.status == 200
        except:
            return True  # assume healthy if no endpoint

class ConcurrencyController:
    def __init__(self, max_workers=10):
        self._sem = asyncio.Semaphore(max_workers)
        self._queue = asyncio.Queue()
        self._workers = []
        self._running = False

    async def start(self, num=5):
        self._running = True
        for _ in range(num):
            self._workers.append(asyncio.create_task(self._worker()))

    async def _worker(self):
        while self._running:
            func, args, kwargs = await self._queue.get()
            async with self._sem:
                try:
                    await func(*args, **kwargs)
                except Exception as e:
                    logger.debug(f"Worker error: {e}")
            self._queue.task_done()

    async def submit(self, func, *args, **kwargs):
        await self._queue.put((func, args, kwargs))

    async def stop(self):
        self._running = False
        for w in self._workers:
            w.cancel()
        await asyncio.gather(*self._workers, return_exceptions=True)
