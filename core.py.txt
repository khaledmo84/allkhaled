@dataclass
class Opportunity:
    id: str = field(default_factory=lambda: f"opp_{int(time.time()*1000)}_{os.urandom(4).hex()}")
    timestamp_ns: int = field(default_factory=time.time_ns)
    parent_id: Optional[str] = None
    name: str = ""
    category: Optional[StrategyCategory] = None
    profit: float = 0.0
    action: str = ""
    params: dict = field(default_factory=dict)
    gas_type: int = GasType.SELF.value
    executor: Optional[ExecutorType] = None
    chain: str = 'ethereum'
    priority: int = 5
    confidence: float = 0.5
    gas_cost: float = 0.0
    tags: List[str] = field(default_factory=list)
    deadline: float = 0.0
    need_approval: bool = True
    amount_in_token: Optional[float] = None
    token_symbol: Optional[str] = None
    token_address: Optional[str] = None
    gas_wei: Optional[int] = None
    required_gas_wei: Optional[int] = None
    ai_generated: bool = False
    flashbots_compatible: bool = False
    supported_agents: List[str] = field(default_factory=list)
    tx_data: Optional[Dict] = None
    target_contract: Optional[str] = None
    call_data: Optional[str] = None
    value: Optional[int] = None
    loan_amount: Optional[int] = None
    loan_token: Optional[str] = None
    loan_source: Optional[str] = None
    path: Optional[List[Dict]] = None
    expected_profit: Optional[int] = None
    min_profit: Optional[int] = None
    execution_contract: Optional[str] = None
    simulation_success: bool = False
    is_safe: bool = False

    def __post_init__(self):
        self.net_profit = self.profit - self.gas_cost
        self.score = self.net_profit * self.confidence * (10 - self.priority + 1)

    def is_expired(self) -> bool:
        return self.deadline > 0 and time.time() > self.deadline

    def to_dict(self) -> Dict:
        result = asdict(self)
        if self.category:
            result['category'] = self.category.value
        if self.executor:
            result['executor'] = self.executor.value
        return result

    @classmethod
    def from_dict(cls, data: Dict) -> 'Opportunity':
        if 'category' in data and isinstance(data['category'], str):
            try:
                data['category'] = StrategyCategory(data['category'])
            except ValueError:
                data['category'] = None
        if 'executor' in data and isinstance(data['executor'], str):
            try:
                data['executor'] = ExecutorType(data['executor'])
            except ValueError:
                data['executor'] = None
        return cls(**data)

                    field_type = cls.__dataclass_fields__[key].type

                        setattr(cfg, key, env_val.lower() in ('true', '1', 'yes'))

                    else:
                        setattr(cfg, key, env_val)

                    setattr(cfg, key, env_val)

def retry(max_retries: int = 3, delay: float = 1.0, backoff: float = 2.0, exceptions: tuple = (Exception,)):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            current_delay = delay
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except exceptions as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        logger.debug(f"Retry {attempt+1}/{max_retries} for {func.__name__}: {e}")
                        await asyncio.sleep(current_delay)
                        current_delay *= backoff
                    else:
                        logger.error(f"All retries failed for {func.__name__}: {e}")
            raise last_exception
        return wrapper
    return decorator


# ==================== ErrorClassifier و smart_retry ====================
class ErrorCategory(Enum):
    NETWORK = "network"
    RPC = "rpc"
    CONTRACT = "contract"
    GAS = "gas"
    NONCE = "nonce"
    UNKNOWN = "unknown"


class ErrorClassifier:
    @staticmethod
    def classify(error: Exception) -> ErrorCategory:
        error_str = str(error).lower()
        if 'connection' in error_str or 'timeout' in error_str or 'socket' in error_str:
            return ErrorCategory.NETWORK
        if 'nonce' in error_str or 'already known' in error_str:
            return ErrorCategory.NONCE
        if 'gas' in error_str or 'intrinsic gas' in error_str:
            return ErrorCategory.GAS
        if 'execution reverted' in error_str or 'invalid opcode' in error_str:
            return ErrorCategory.CONTRACT
        if 'rpc' in error_str or 'http' in error_str:
            return ErrorCategory.RPC
        return ErrorCategory.UNKNOWN


def smart_retry(max_retries: int = 3, delay: float = 1.0, backoff: float = 2.0):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            current_delay = delay
            rpc_manager = kwargs.get('rpc_manager')
            nonce_manager = kwargs.get('nonce_manager')
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    category = ErrorClassifier.classify(e)
                    if category == ErrorCategory.CONTRACT:
                        logger.error(f"Unrecoverable contract error: {e}")
                        raise
                    if category == ErrorCategory.RPC and rpc_manager:
                        await rpc_manager.switch_node()
                    if category == ErrorCategory.NONCE and nonce_manager:
                        await nonce_manager.reset()
                    if attempt < max_retries - 1:
                        logger.warning(f"Retry {attempt+1}/{max_retries} for {func.__name__} due to {category.value}: {e}")
                        await asyncio.sleep(current_delay)
                        current_delay *= backoff
                    else:
                        logger.error(f"All retries failed for {func.__name__}: {e}")
            raise last_exception
        return wrapper
    return decorator

class MABAgent:
    def __init__(self, exp_db: 'ExperienceDB', epsilon: float = 0.1):
        self.exp_db = exp_db
        self.epsilon = epsilon
        self.arms = defaultdict(lambda: {'successes': 0, 'trials': 0})

    async def select_arm(self, available_agents: List[str]) -> str:
        if random.random() < self.epsilon:
            return random.choice(available_agents)
        best_agent = None
        best_ucb = -float('inf')
        total_trials = sum(self.arms.get(a, {}).get('trials', 0) for a in available_agents)
        for agent in available_agents:
            stats = self.arms.get(agent, {'successes': 0, 'trials': 0})
            if stats['trials'] == 0:
                return agent
            avg_reward = stats['successes'] / stats['trials']
            exploration = math.sqrt(2 * math.log(total_trials + 1) / stats['trials'])
            ucb = avg_reward + exploration
            if ucb > best_ucb:
                best_ucb = ucb
                best_agent = agent
        return best_agent or random.choice(available_agents)

    async def update_arm(self, agent_name: str, reward: float):
        if agent_name not in self.arms:
            self.arms[agent_name] = {'successes': 0, 'trials': 0}
        self.arms[agent_name]['trials'] += 1
        if reward > 0:
            self.arms[agent_name]['successes'] += 1

class AESGCMEncryptor:
    def __init__(self, password: str, salt: Optional[bytes] = None):
        if not CRYPTO_AVAILABLE:
            raise ImportError("cryptography library required")
        self.password = password.encode()
        self.salt = salt or os.urandom(16)
        self._derive_key()

    def _derive_key(self, iterations: int = 200000):
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=self.salt,
            iterations=iterations,
            backend=default_backend()
        )
        self.key = kdf.derive(self.password)
        self.aesgcm = AESGCM(self.key)

    def encrypt(self, data: bytes) -> bytes:
        nonce = os.urandom(12)
        ciphertext = self.aesgcm.encrypt(nonce, data, None)
        return self.salt + nonce + ciphertext

    def decrypt(self, encrypted_data: bytes) -> bytes:
        salt = encrypted_data[:16]
        nonce = encrypted_data[16:28]
        ciphertext = encrypted_data[28:]
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=200000,
            backend=default_backend()
        )
        key = kdf.derive(self.password)
        aesgcm = AESGCM(key)
        return aesgcm.decrypt(nonce, ciphertext, None)


class SecureRedis:
    def __init__(self, redis_url: str, encryptor: Optional[AESGCMEncryptor] = None):
        if not REDIS_AVAILABLE:
            raise ImportError("redis.asyncio required")
        self.redis = redis.from_url(redis_url, decode_responses=False)
        self.encryptor = encryptor

    async def set(self, key: str, value: bytes, ex: Optional[int] = None):
        if self.encryptor:
            value = self.encryptor.encrypt(value)
        await self.redis.set(key, value, ex=ex)

    async def get(self, key: str) -> Optional[bytes]:
        value = await self.redis.get(key)
        if value and self.encryptor:
            value = self.encryptor.decrypt(value)
        return value

    async def delete(self, key: str):
        await self.redis.delete(key)

    async def incr(self, key: str) -> int:
        return await self.redis.incr(key)

    async def setnx(self, key: str, value: bytes) -> bool:

    async def expire(self, key: str, seconds: int):

        ]

        self.cache = {}
        self.validation_funcs = {
            'coingecko': self._validate_coingecko,
            'etherscan': self._validate_etherscan,
            'infura': self._validate_infura,
            'alchemy': self._validate_alchemy,
        }

    async def _validate_coingecko(self, key: str) -> bool:
        async with aiohttp.ClientSession() as session:
            try:
                async with session.get(f"https://api.coingecko.com/api/v3/ping?x_cg_pro_api_key={key}", timeout=5) as resp:
                    return resp.status == 200
            except:
                return False

    async def _validate_etherscan(self, key: str) -> bool:
        async with aiohttp.ClientSession() as session:
            try:
                async with session.get(f"https://api.etherscan.io/api?module=proxy&action=eth_blockNumber&apikey={key}", timeout=5) as resp:
                    data = await resp.json()
                    return data.get('status') == '1'
            except:
                return False

    async def _validate_infura(self, key: str) -> bool:
        async with aiohttp.ClientSession() as session:
            try:
                async with session.post(f"https://mainnet.infura.io/v3/{key}", json={"jsonrpc":"2.0","method":"eth_blockNumber","params":[],"id":1}, timeout=5) as resp:
                    data = await resp.json()
                    return 'result' in data
            except:
                return False

    async def _validate_alchemy(self, key: str) -> bool:
        async with aiohttp.ClientSession() as session:
            try:
                async with session.post(f"https://eth-mainnet.g.alchemy.com/v2/{key}", json={"jsonrpc":"2.0","method":"eth_blockNumber","params":[],"id":1}, timeout=5) as resp:
                    data = await resp.json()
                    return 'result' in data
            except:
                return False

    async def get_key(self, source: str) -> Optional[str]:
        if source not in self.sources:
            return None
        if self.use_public_apis and source in self.public_sources:
            return None
        if source in self.cache:
            key, expiry = self.cache[source]
            if time.time() < expiry:
                return key
        keys_data = await self.key_manager.get_key(f"apikeys:{source}")
        if not keys_data:
            return None
        try:
            keys = json.loads(keys_data.decode())
        except:
            keys = []
        validate = self.validation_funcs.get(source, lambda k: True)
        for key in keys:
            if await validate(key):
                self.cache[source] = (key, time.time() + 300)
                return key
        return None

    async def set_keys(self, source: str, keys: List[str]):
        await self.key_manager.set_key(f"apikeys:{source}", json.dumps(keys).encode())

    async def check_all_keys(self) -> Dict[str, bool]:
        results = {}
        for source in self.sources:
            keys_data = await self.key_manager.get_key(f"apikeys:{source}")
            if not keys_data:
                results[source] = False
                continue
            try:
                keys = json.loads(keys_data.decode())
                validate = self.validation_funcs.get(source, lambda k: True)
                valid = any(await validate(k) for k in keys)
                results[source] = valid
            except:
                results[source] = False
        return results

class KeyMaintenanceAgent(BaseMaintenanceAgent):

            self._increment_failure(str(e), "key_maintenance")
            await self.notification_queue.put(f"? Key maintenance failed: {e}")
        return results

class AtomicNonceManager:
    def __init__(self, redis: Optional[SecureRedis], address: str, chain_id: int, use_sqlite: bool = True):
        self.redis = redis
        self.address = address.lower()
        self.chain_id = chain_id
        self.key = f"nonce:{self.address}:{chain_id}"
        self.local_nonce: Optional[int] = None
        self.lock = asyncio.Lock()
        self.reserved_nonces: Set[int] = set()
        self.max_attempts = 10
        self.use_sqlite = use_sqlite and not redis
        self.sqlite_db = None
        self._lua_script = """
            local key = KEYS[1]
            local chain_nonce = tonumber(ARGV[1])
            local current = redis.call('GET', key)
            if not current then
                redis.call('SET', key, chain_nonce)
                return chain_nonce
            else
                current = tonumber(current)
                if current < chain_nonce then
                    redis.call('SET', key, chain_nonce)
                    current = chain_nonce
                end
                local new_nonce = current + 1
                redis.call('SET', key, new_nonce)
                return new_nonce
            end
        """ if REDIS_AVAILABLE else None

    async def _get_sqlite(self):
        if not self.sqlite_db:
            self.sqlite_db = await aiosqlite.connect('nonce.db')
            await self.sqlite_db.execute('CREATE TABLE IF NOT EXISTS nonces (key TEXT PRIMARY KEY, nonce INTEGER)')
            await self.sqlite_db.commit()
        return self.sqlite_db

    async def get_nonce(self, w3: AsyncWeb3) -> int:
        async with self.lock:
            chain_nonce = await w3.eth.get_transaction_count(self.address, 'pending')
            if self.redis:
                try:
                    if hasattr(self.redis, 'redis') and self.redis.redis:
                        new_nonce = await self.redis.redis.eval(
                            self._lua_script,
                            1,
                            self.key,
                            str(chain_nonce)
                        )
                        new_nonce = int(new_nonce)
                    else:
                        success = await self.redis.setnx(self.key, str(chain_nonce))
                        if success:
                            new_nonce = chain_nonce
                        else:
                            new_nonce = await self.redis.incr(self.key)
                            if new_nonce < chain_nonce:
                                await self.redis.set(self.key, str(chain_nonce))
                                new_nonce = chain_nonce
                except Exception:
                    success = await self.redis.setnx(self.key, str(chain_nonce))
                    if success:
                        new_nonce = chain_nonce
                    else:
                        new_nonce = await self.redis.incr(self.key)
                        if new_nonce < chain_nonce:
                            await self.redis.set(self.key, str(chain_nonce))
                            new_nonce = chain_nonce
                attempts = 0
                while new_nonce in self.reserved_nonces and attempts < self.max_attempts:
                    new_nonce = await self.redis.incr(self.key)
                    attempts += 1
                if attempts >= self.max_attempts:
                    raise Exception("Too many nonce collisions")
                self.reserved_nonces.add(new_nonce)
                return new_nonce
            elif self.use_sqlite:
                db = await self._get_sqlite()
                async with db.execute('SELECT nonce FROM nonces WHERE key = ?', (self.key,)) as cursor:
                    row = await cursor.fetchone()
                if row:
                    current = row[0]
                    if current < chain_nonce:
                        new_nonce = chain_nonce
                    else:
                        new_nonce = current + 1
                else:
                    new_nonce = chain_nonce
                await db.execute('INSERT OR REPLACE INTO nonces (key, nonce) VALUES (?, ?)', (self.key, new_nonce))
                await db.commit()
                attempts = 0
                while new_nonce in self.reserved_nonces and attempts < self.max_attempts:
                    new_nonce += 1
                    await db.execute('UPDATE nonces SET nonce = ? WHERE key = ?', (new_nonce, self.key))
                    await db.commit()
                    attempts += 1
                if attempts >= self.max_attempts:
                    raise Exception("Too many nonce collisions")
                self.reserved_nonces.add(new_nonce)
                return new_nonce
            else:
                if self.local_nonce is None or self.local_nonce < chain_nonce:
                    self.local_nonce = chain_nonce
                else:
                    self.local_nonce += 1
                attempts = 0
                while self.local_nonce in self.reserved_nonces and attempts < self.max_attempts:
                    self.local_nonce += 1
                    attempts += 1
                if attempts >= self.max_attempts:
                    raise Exception("Too many nonce collisions")
                self.reserved_nonces.add(self.local_nonce)
                return self.local_nonce

    async def commit_nonce(self, nonce: int):
        if self.redis:
            current = await self.redis.get(self.key)
            if current and int(current) <= nonce:
                await self.redis.set(self.key, str(nonce + 1))
        elif self.use_sqlite:
            db = await self._get_sqlite()
            await db.execute('UPDATE nonces SET nonce = ? WHERE key = ?', (nonce + 1, self.key))
            await db.commit()
        else:
            if self.local_nonce and self.local_nonce <= nonce:
                self.local_nonce = nonce + 1
        self.reserved_nonces.discard(nonce)

    async def reset(self):
        self.reserved_nonces.clear()
        if self.redis:
            await self.redis.delete(self.key)
        elif self.use_sqlite:
            db = await self._get_sqlite()
            await db.execute('DELETE FROM nonces WHERE key = ?', (self.key,))
            await db.commit()


# ==================== RateLimiter ====================
class RateLimiter:
    def __init__(self, redis: Optional[SecureRedis], max_requests: int, window_seconds: int, name: str = "default"):
        self.redis = redis
        self.max_requests = max_requests
        self.window = window_seconds
        self.name = name
        self.local_counter = defaultdict(list)
        self._cleanup_task: Optional[asyncio.Task] = None

    async def start_cleanup(self):
        async def cleanup():
            while True:
                await asyncio.sleep(60)
                now = time.time()
                for key in list(self.local_counter.keys()):
                    self.local_counter[key] = [t for t in self.local_counter[key] if t > now - self.window]
                    if not self.local_counter[key]:
                        del self.local_counter[key]
        self._cleanup_task = asyncio.create_task(cleanup())

    async def acquire(self, key: str) -> bool:
        now = time.time()
        full_key = f"{self.name}:{key}"
        if self.redis:
            window_start = now - self.window
            redis_key = f"ratelimit:{full_key}"
            await self.redis.zremrangebyscore(redis_key, 0, window_start)
            count = await self.redis.zcard(redis_key)
            if count < self.max_requests:
                member = str(uuid.uuid4())
                await self.redis.zadd(redis_key, {member: now})
                await self.redis.expire(redis_key, self.window + 10)
                return True
            return False
        else:
            self.local_counter[full_key] = [t for t in self.local_counter[full_key] if t > now - self.window]
            if len(self.local_counter[full_key]) < self.max_requests:
                self.local_counter[full_key].append(now)
                return True
            return False

    async def stop_cleanup(self):
        if self._cleanup_task:
            self._cleanup_task.cancel()
            try:
                await self._cleanup_task
            except asyncio.CancelledError:
                pass


# ==================== Quantum Cache ====================
class QuantumCache:
    def __init__(self, name: str, size: int = 1024 * 1024 * 100, max_entries: int = 10000):
        self.name = name
        self.size = size
        self.max_entries = max_entries
        self.mmap = None
        self.lock = asyncio.Lock()
        self.index: Dict[str, Tuple[int, int, float]] = {}
        self.lru: deque = deque(maxlen=max_entries)
        self.write_pos = 0
        self._init_mmap()
        self._cleanup_task: Optional[asyncio.Task] = None

    def _init_mmap(self):
        try:
            self.mmap = mmap.mmap(-1, self.size, self.name)
        except Exception as e:
            logger.error(f"Failed to create mmap {self.name}: {e}")
            self.mmap = None

    async def set(self, key: str, value: Any, ttl: Optional[int] = None):
        if not self.mmap:
            return
        async with self.lock:
            try:
                data = msgpack.packb({'key': key, 'value': value, 'expires': time.time() + ttl if ttl else 0})
                compressed = compress_zstd(data)
                length = len(compressed)
                if length + 8 > self.size:
                    logger.warning(f"Value too large for QuantumCache: {length} bytes")
                    return

                if key in self.index:
                    old_pos, old_len, _ = self.index[key]
                    del self.index[key]
                    if key in self.lru:
                        self.lru.remove(key)

                if self.write_pos + 4 + length > self.size:
                    self.write_pos = 0
                pos = self.write_pos
                self.mmap.seek(pos)
                self.mmap.write(struct.pack('I', length))
                self.mmap.write(compressed)
                self.write_pos += 4 + length

                expires = time.time() + ttl if ttl else 0
                self.index[key] = (pos, length, expires)
                self.lru.append(key)

                if len(self.index) > self.max_entries:
                    oldest = self.lru.popleft()
                    if oldest in self.index:
                        del self.index[oldest]
            except Exception as e:
                logger.error(f"QuantumCache set error: {e}")

    async def get(self, key: str) -> Optional[Any]:
        if not self.mmap:
            return None
        async with self.lock:
            entry = self.index.get(key)
            if not entry:
                return None
            pos, length, expires = entry
            if expires and expires < time.time():
                del self.index[key]
                if key in self.lru:
                    self.lru.remove(key)
                return None
            try:
                self.mmap.seek(pos)
                len_data = self.mmap.read(4)
                if len(len_data) < 4:
                    return None
                stored_len = struct.unpack('I', len_data)[0]
                if stored_len != length:
                    return None
                data = self.mmap.read(length)
                decompressed = decompress_zstd(data)
                obj = msgpack.unpackb(decompressed)
                if key in self.lru:
                    self.lru.remove(key)
                self.lru.append(key)
                return obj['value']
            except Exception as e:
                logger.error(f"QuantumCache get error: {e}")
                return None

    async def start_cleanup(self, interval: float = 60.0):
        async def cleanup():
            while True:
                await asyncio.sleep(interval)
                await self._cleanup_expired()
        self._cleanup_task = asyncio.create_task(cleanup())

    async def _cleanup_expired(self):
        now = time.time()
        expired = [k for k, (_, _, exp) in self.index.items() if exp and exp < now]
        for k in expired:
            del self.index[k]
            if k in self.lru:
                self.lru.remove(k)

    async def close(self):
        if self._cleanup_task:
            self._cleanup_task.cancel()
            try:
                await self._cleanup_task
            except asyncio.CancelledError:
                pass
        if self.mmap:
            self.mmap.close()


# ==================== SmartCache ====================
class SmartCache:
    def __init__(self, redis: Optional[SecureRedis] = None, quantum: Optional[QuantumCache] = None,
                 max_size: int = 10000, default_ttl: int = 10):
        self.redis = redis
        self.quantum = quantum
        self.default_ttl = default_ttl
        self.max_size = max_size
        self.memory: OrderedDict = OrderedDict()
        self.memory_ttl: Dict[str, float] = {}
        self.lock = asyncio.Lock()

    async def get(self, key: str) -> Optional[Any]:
        async with self.lock:
            if key in self.memory:
                if time.time() < self.memory_ttl.get(key, 0):
                    self.memory.move_to_end(key)
                    return self.memory[key]
                else:
                    del self.memory[key]
                    del self.memory_ttl[key]
        if self.quantum:
            val = await self.quantum.get(key)
            if val:
                await self._set_memory(key, val, self.default_ttl)
                return val
        if self.redis:
            data = await self.redis.get(f"cache:{key}")
            if data:
                data = decompress_zstd(data)
                value = pickle.loads(data)
                await self._set_memory(key, value, self.default_ttl)
                return value
        return None

    async def set(self, key: str, value: Any, ttl: Optional[int] = None):
        ttl = ttl or self.default_ttl
        await self._set_memory(key, value, ttl)
        if self.quantum:
            await self.quantum.set(key, value, ttl)
        if self.redis:
            data = pickle.dumps(value)
            data = compress_zstd(data)
            await self.redis.set(f"cache:{key}", data, ex=ttl)

    async def _set_memory(self, key: str, value: Any, ttl: int):
        async with self.lock:
            now = time.time()
            expired = [k for k, exp in list(self.memory_ttl.items()) if exp < now]
            for k in expired:
                del self.memory[k]
                del self.memory_ttl[k]
            if key in self.memory:
                self.memory.move_to_end(key)
            else:
                if len(self.memory) >= self.max_size:
                    self.memory.popitem(last=False)
            self.memory[key] = value
            self.memory_ttl[key] = time.time() + ttl

    async def delete(self, key: str):
        async with self.lock:
            self.memory.pop(key, None)
            self.memory_ttl.pop(key, None)
        if self.redis:
            await self.redis.delete(f"cache:{key}")

    async def clear(self):
        async with self.lock:
            self.memory.clear()
            self.memory_ttl.clear()

    async def cleanup_expired(self):
        now = time.time()
        async with self.lock:
            expired = [k for k, t in self.memory_ttl.items() if t < now]
            for k in expired:
                del self.memory[k]
                del self.memory_ttl[k]


# ==================== AdaptiveCache ====================
class AdaptiveCache(SmartCache):
    def __init__(self, redis=None, quantum=None, max_size=10000, default_ttl=10, adaptation_factor=0.1):
        super().__init__(redis, quantum, max_size, default_ttl)
        self.adaptation_factor = adaptation_factor
        self.hit_rates = defaultdict(lambda: deque(maxlen=100))
        self.last_ttl = {}

    async def get(self, key: str) -> Optional[Any]:
        start = time.time()
        value = await super().get(key)
        if value:
            self.hit_rates[key].append(1)
            if len(self.hit_rates[key]) > 50 and sum(self.hit_rates[key])/len(self.hit_rates[key]) > 0.9:
                current_ttl = self.last_ttl.get(key, self.default_ttl)
                new_ttl = min(current_ttl * (1 + self.adaptation_factor), 300)
                self.last_ttl[key] = new_ttl
        else:
            self.hit_rates[key].append(0)
        return value

            latency = time.time() - start

        if not self.is_healthy:
            return float('inf')
        latency_score = self.latency
        reliability = self.successes / max(1, self.successes + self.failures)
        return latency_score * (2 - reliability)

        return self.predictor.should_switch()

        await self.session.close()


class PredictiveFailureDetector:
    def __init__(self, history_size: int = 100, threshold: float = 2.0):
        self.history = deque(maxlen=history_size)
        self.threshold = threshold
        self.predictions = deque(maxlen=10)
        self.failure_probability = 0.0
        self.alpha = 0.3
        self.mean = None
        self.variance = None

    def update(self, latency: float):
        self.history.append(latency)
        self._predict_ewma()

    def _predict_ewma(self):
        if len(self.history) < 10:
            return
        if self.mean is None:
            self.mean = np.mean(self.history)
            self.variance = np.var(self.history)
        else:
            new_mean = self.alpha * self.history[-1] + (1 - self.alpha) * self.mean
            new_variance = self.alpha * (self.history[-1] - new_mean)**2 + (1 - self.alpha) * self.variance
            self.mean = new_mean
            self.variance = new_variance
        forecast = self.mean
        self.predictions.append(forecast)
        if len(self.predictions) > 3:
            avg_pred = np.mean(self.predictions)
            if self.mean > 0:
                self.failure_probability = min(1.0, avg_pred / (self.mean * self.threshold))

    def should_switch(self) -> bool:
        if len(self.predictions) < 3:
            return False
        avg_pred = np.mean(self.predictions)
        return avg_pred > self.mean * self.threshold if self.mean else False
