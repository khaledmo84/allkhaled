class PriceFetcher:
    """جلب الأسعار من CoinGecko و RPC"""
    def __init__(self, config: Config, cache: SmartCache, god_pulse: Dict[str, GodPulse]):
        self.config = config
        self.cache = cache
        self.god_pulse = god_pulse
        self.session = None
        self.coingecko_api_key = config.coingecko_api_key
        self.api_key_manager = None  # يربط من الخارج

    async def _ensure_session(self):
        if not self.session:
            self.session = aiohttp.ClientSession()

        if api_key:
            url += f"&x_cg_pro_api_key={api_key}"
        try:
            async with self.session.get(url, timeout=5) as resp:
                if resp.status == 200:
                    data = await resp.json()
                    price = float(data['ethereum']['usd'])
                    await self.cache.set(cache_key, price, ttl=60)
                    return price
        except Exception as e:
            logger.warning(f"CoinGecko ETH price failed: {e}")

        # الرجوع إلى RPC (محاولة الحصول من Uniswap)
        try:
            gp = self.god_pulse.get(chain)
            if gp:
                node = await gp.get_best_node()
                if node:
                    # استخدام عقدة للحصول على سعر ETH/USDC من Uniswap (مثال)
                    # هذا يتطلب استدعاء عقد، نتركه مؤقتاً
                    pass
        except:
            pass

        # قيمة افتراضية
        return 2000.0

class MempoolWatcher:
    """مراقبة الميمبول عبر WebSocket أو RPC"""
    def __init__(self, god_pulse: GodPulse, config: Config, ws_manager: WebSocketManager):
        self.god_pulse = god_pulse
        self.config = config
        self.ws_manager = ws_manager
        self.pending_transactions: List[Dict] = []
        self.running = False
        self._task: Optional[asyncio.Task] = None

    async def start(self):
        self.running = True
        self._task = asyncio.create_task(self._watch())

    async def _watch(self):
        # يمكن استخدام WebSocket إذا كان متاحاً، وإلا نستخدم polling (غير مدعوم)
        # هنا سنستخدم محاكاة مؤقتة
        while self.running:
            # محاكاة وصول معاملة
            if random.random() < 0.1:
                tx = {
                    'hash': '0x' + os.urandom(32).hex(),
                    'from': '0x' + os.urandom(20).hex(),
                    'to': '0x' + os.urandom(20).hex(),
                    'value': random.randint(1, 100) * 10**18,
                    'gasPrice': random.randint(10, 200) * 10**9,
                    'input': '0x' + os.urandom(100).hex()
                }
                self.pending_transactions.append(tx)
                if len(self.pending_transactions) > 1000:
                    self.pending_transactions = self.pending_transactions[-1000:]
            await asyncio.sleep(1)

    def get_swaps_in_mempool(self, limit: int = 10) -> List[Dict]:
        swaps = [tx for tx in self.pending_transactions if tx.get('input', '').startswith(('0x7ff36ab5', '0x18cbafe5', '0x5c11d795'))]
        return swaps[-limit:]

    async def stop(self):
        self.running = False
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass

    async def _fetch_uniswap_prices(self, chain: str) -> Dict[str, float]:
        if chain != 'ethereum':
            return {}
        endpoint = "https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v2"
        api_key = None
        if self.api_key_manager:
            api_key = await self.api_key_manager.get_key('thegraph')
        headers = {}
        if api_key:
            headers['Authorization'] = f'Bearer {api_key}'
        query = """
        {
            tokens(first: 50, orderBy: derivedETH, orderDirection: desc) {
                id
                symbol
                derivedETH
            }
        }
        """
        try:
            async with self.session.post(endpoint, json={'query': query}, headers=headers) as resp:
                if resp.status == 200:
                    data = await resp.json()
                    tokens = data.get('data', {}).get('tokens', [])
                    eth_price = await self.price_fetcher.get_eth_price(chain)
                    prices = {}
                    for t in tokens:
                        try:
                            price_in_eth = float(t['derivedETH'])
                            prices[t['id']] = price_in_eth * eth_price
                        except:
                            continue
                    return prices
        except Exception as e:
            logger.warning(f"Uniswap TheGraph query failed: {e}")
        return {}

    async def _fetch_lending_positions(self, chain: str) -> List[Dict]:
        if chain != 'ethereum':
            return []
        endpoint = "https://api.thegraph.com/subgraphs/name/aave/protocol-v3"
        api_key = None
        if self.api_key_manager:
            api_key = await self.api_key_manager.get_key('thegraph')
        headers = {}
        if api_key:
            headers['Authorization'] = f'Bearer {api_key}'
        query = """
        {
            users(first: 100, where: {borrowedReservesCount_gt: 0}) {
                id
                reserves {
                    currentATokenBalance
                    currentVariableDebt
                }
                healthFactor
            }
        }
        """
        try:
            async with self.session.post(endpoint, json={'query': query}, headers=headers) as resp:
                if resp.status == 200:
                    data = await resp.json()
                    positions = []
                    for user in data.get('data', {}).get('users', []):
                        try:
                            health = float(user['healthFactor'])
                            if health < 1.1:
                                total_debt = sum(float(r['currentVariableDebt']) for r in user['reserves']) / 1e18
                                total_collateral = sum(float(r['currentATokenBalance']) for r in user['reserves']) / 1e18
                                positions.append({
                                    'user': user['id'],
                                    'debt': total_debt,
                                    'collateral': total_collateral,
                                    'health': health,
                                    'protocol': 'Aave V3'
                                })
                        except:
                            continue
                    return positions
        except Exception as e:
            logger.warning(f"Aave TheGraph query failed: {e}")
        return []

        self.execution_queue = asyncio.Queue()
        self.stats = defaultdict(int)
        self.stats_lock = asyncio.Lock()
        self.semaphore = asyncio.Semaphore(config.max_concurrent_executions)
        self.worker_task: Optional[asyncio.Task] = None

    async def start_worker(self):
        self.worker_task = asyncio.create_task(self._worker())

    async def _worker(self):
        while True:
            opp = await self.execution_queue.get()
            async with self.semaphore:
                try:
                    await self._execute_single(opp)
                except Exception as e:
                    logger.error(f"Execution worker error: {e}")
            self.execution_queue.task_done()

        # تنفيذ فرصة (موجود في core)

        if key not in self.nonce_managers:
            self.nonce_managers[key] = AtomicNonceManager(self.redis, self.signer.address, chain_id, use_sqlite=True)

# ==================== DataQualityAgent ====================
class DataQualityAgent(BaseMaintenanceAgent):
    def __init__(self, config: MaintenanceConfig, market_aggregator: MarketDataAggregator):
        super().__init__("DataQuality", config)
        self.market_aggregator = market_aggregator
        self.quality_scores = defaultdict(float)

    async def run(self) -> Dict[str, Any]:
        self.logger.info("Checking data quality...")
        results = {}
        try:
            for chain in ['ethereum', 'bsc', 'polygon']:
                md = await self.market_aggregator.get_snapshot(chain)
                score = self._calculate_quality(md)
                self.quality_scores[chain] = score
                results[chain] = score
                if score < 0.5:
                    await self.notification_queue.put(f"📊 Low data quality on {chain}: {score:.2f}")
            self._increment_success("data_quality")
        except Exception as e:
            self._increment_failure(str(e), "data_quality")
            await self.notification_queue.put(f"❌ Data quality check failed: {e}")
        return results

    def _calculate_quality(self, md: MarketData) -> float:
        score = 1.0
        if not md.eth_price or md.eth_price < 100:
            score -= 0.3
        if not md.gas_gwei or md.gas_gwei < 1:
            score -= 0.2
        if not md.uniswap_prices or len(md.uniswap_prices) < 5:
            score -= 0.2
        if not md.liquidity_depth or len(md.liquidity_depth) < 5:
            score -= 0.2
        return max(0, score)

            'requests', 'schedule', 'libp2p', 'pymysql', 'psycopg2',
            'boto3', 'amazon-braket-sdk', 'tenseal', 'ape-solidity',
            'langgraph', 'pyautogen', 'crewai', 'composio-core',
            'browser-use', 'injective-py', 'ragas', 'gpt-researcher'
        ]

    async def check_versions(self) -> Dict[str, bool]:
        results = {}
        for pkg in self.required_packages + self.optional_packages:
            try:
                __import__(pkg.replace('-', '_'))
                results[pkg] = True
            except ImportError:
                results[pkg] = False
        return results

    async def install_missing(self, packages: List[str]) -> bool:
        try:
            for pkg in packages:
                proc = await asyncio.create_subprocess_exec(
                    sys.executable, '-m', 'pip', 'install', pkg,
                    stdout=asyncio.PIPE, stderr=asyncio.PIPE
                )
                await proc.communicate()
            return True
        except Exception as e:
            logger.error(f"Failed to install packages: {e}")
            return False

    async def _opportunity_processor(self):
        while self.running and not self.emergency_stop:
            try:
                opp = await self.detector.get_opportunity(timeout=1.0)
                if not opp:
                    await asyncio.sleep(0.1)
                    continue

                async with self.stats_lock:
                    self.stats['opportunities_received'] += 1

                if self.emergency_stop:
                    continue

                market_data = await self.market_aggregator.get_snapshot(opp.chain)
                risk_ok, msg = await self.risk_manager.check(opp, market_data=market_data)
                if not risk_ok:
                    logger.debug(f"Risk check failed: {msg}")
                    continue

                if not await self.circuit_breaker.check(opp.amount_in_token or 0):
                    continue

                new_opp, confidence, data = await self.orchestrator.orchestrate(opp, market_data)

                if new_opp and confidence >= self.config.min_confidence:
                    await self.exp_db.record_opportunity(new_opp)
                    if new_opp.profit >= self.config.min_profit:
                        async with self.stats_lock:
                            self.stats['opportunities_executed'] += 1
                        await self.worker_pool.submit(self._execute_opportunity, new_opp)

            except Exception as e:
                logger.error(f"Opportunity processor error: {e}")
                traceback.print_exc()
                await asyncio.sleep(1)

    async def _execute_opportunity(self, opportunity: Opportunity):
        if self.emergency_stop:
            return
        result = await self.execution_engine.execute_opportunity(opportunity)

        if result:
            async with self.stats_lock:
                self.stats['successful_executions'] += 1
                self.stats['total_profit'] += opportunity.profit
            await self.risk_manager.update(opportunity.profit, opportunity.token_symbol, opportunity.amount_in_token)
            await self.circuit_breaker.record_trade(opportunity.profit)

            trade = TradeRecord(
                id=f"trade_{int(time.time()*1000)}_{os.urandom(4).hex()}",
                opportunity_id=opportunity.id,
                timestamp_ns=time_now_ns(),
                chain=opportunity.chain,
                executor=opportunity.executor.value if opportunity.executor else 'unknown',
                profit_usd=opportunity.profit,
                gas_cost_usd=opportunity.gas_cost,
                tx_hash=result.get('tx_hash') if isinstance(result, dict) else None,
                metadata={'result': result}
            )
            await self.exp_db.record_trade(trade)

            if self.reinvestor and opportunity.profit > 0:
                await self.reinvestor.add_profit(opportunity.profit)

            if self.telegram:
                msg = f"✅ Executed {opportunity.name} | Profit: ${opportunity.profit:.2f}"
                await self.telegram.notify(msg)
        else:
            async with self.stats_lock:
                self.stats['failed_executions'] += 1
            await self.exp_db.record_failure(
                opportunity.executor.value if opportunity.executor else 'unknown',
                opportunity.chain,
                "Execution failed",
                opportunity.profit
            )

    def get_stats(self) -> Dict:
        async def _get_stats():
            uptime = time_elapsed_ns(self.start_time_ns) / 1e9
            async with self.stats_lock:
                stats_copy = dict(self.stats)
            orchestrator_stats = self.orchestrator.get_stats()
            detector_stats = self.detector.get_stats()
            exec_stats = await self.execution_engine.get_stats()
            risk_report = await self.risk_manager.get_risk_report()
            maint_stats = self.maintenance_manager.get_stats() if self.maintenance_manager else {}
            data_quality = self.data_quality_agent.quality_scores if self.data_quality_agent else {}
            thinking_stats = self.thinking.last_analysis if self.thinking else {}
            return {
                'uptime_sec': uptime,
                'stats': stats_copy,
                'orchestrator': orchestrator_stats,
                'detector': detector_stats,
                'execution': exec_stats,
                'risk': risk_report,
                'maintenance': maint_stats,
                'data_quality': data_quality,
                'thinking': thinking_stats
            }
        try:
            loop = asyncio.get_running_loop()
        except RuntimeError:
            return {
                'uptime_sec': time_elapsed_ns(self.start_time_ns) / 1e9,
                'stats': dict(self.stats),
                'orchestrator': self.orchestrator.get_stats(),
                'detector': self.detector.get_stats(),
                'execution': {},
                'risk': {}
            }
        future = asyncio.run_coroutine_threadsafe(_get_stats(), loop)
        return future.result(timeout=5) if loop.is_running() else {}

    async def process_query(self, query: str) -> Dict:
        start_ns = time_now_ns()
        try:
            # التحقق مما إذا كان الأمر موجهاً للعقل المدبر
            if self.thinking and query.strip().startswith(('فكر', 'حلل', 'اقتراح', 'طبق', 'استشر', 'نصيحة', 'مساعدة')):
                result = await self.thinking.execute_arabic(query)
                return {
                    'status': 'success',
                    'query': query,
                    'response': result,
                    'processing_time_ns': time_elapsed_ns(start_ns)
                }
            else:
                opp = Opportunity(name=query, category=StrategyCategory.OTHER, confidence=0.5)
                market_data = await self.market_aggregator.get_snapshot('ethereum')
                new_opp, confidence, data = await self.orchestrator.orchestrate(opp, market_data)
                async with self.stats_lock:
                    self.stats['queries_processed'] += 1
                return {
                    'status': 'success',
                    'query': query,
                    'opportunity': new_opp.to_dict() if new_opp else None,
                    'confidence': confidence,
                    'data': data,
                    'processing_time_ns': time_elapsed_ns(start_ns)
                }
        except Exception as e:
            async with self.stats_lock:
                self.stats['query_errors'] += 1
            return {
                'status': 'error',
                'query': query,
                'error': str(e),
                'processing_time_ns': time_elapsed_ns(start_ns)
            }

async def shutdown(loop):
    tasks = [t for t in asyncio.all_tasks() if t is not asyncio.current_task()]
    for t in tasks:
        t.cancel()
    await asyncio.gather(*tasks, return_exceptions=True)
    loop.stop()

def handle_signals():
    loop = asyncio.get_event_loop()
    for sig in (signal.SIGINT, signal.SIGTERM):
        loop.add_signal_handler(sig, lambda: asyncio.create_task(shutdown(loop)))

        logger.info("AlKhaled Ultimate Agent is running")
        logger.info(f"Configuration: {json.dumps(asdict(config), indent=2, default=str)}")

        if len(sys.argv) > 1 and sys.argv[1] == "--api":
            if FASTAPI_AVAILABLE:
                logger.info("Starting FastAPI server on http://0.0.0.0:8000")
                uvicorn.run(app, host="0.0.0.0", port=8000)
            else:
                logger.error("FastAPI not available")
        elif len(sys.argv) > 1 and sys.argv[1] == "--streamlit":
            run_streamlit()
        else:
            logger.info("Interactive mode. Type 'exit' to quit.")
            while True:
                query = input("\nEnter query: ").strip()
                if query.lower() in ['exit', 'quit']:
                    break
                if query:
                    result = await agent.process_query(query)
                    print(json.dumps(result, indent=2, ensure_ascii=False))
    except KeyboardInterrupt:
        logger.info("Shutdown requested")
    finally:
        await agent.stop()
        logger.info("AlKhaled Ultimate Agent stopped")

if __name__ == "__main__":
    handle_signals()
    asyncio.run(main())
