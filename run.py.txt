#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
تشغيل البوت مباشرة مع تطبيق جميع التصحيحات تلقائياً
"""
import asyncio
import sys
import os

# تطبيق التصحيحات تلقائياً قبل أي استيراد آخر
from foundation import AsyncSQLiteConnector, DistributedIdempotencyKeys, MemoryGuard, MultiVersionRollback, HealthCheckServer, GracefulShutdown, CrashReporter
from manager import ExtendedToolsManager

# تصحيح sqlite3
sqlite_patch = AsyncSQLiteConnector()
sqlite_patch.patch()

# استيراد الوكيل الأصلي (unified.py)
from unified import AlKhaledUltimateAgent
from core import Config

async def main():
    config = Config.from_env()
    agent = AlKhaledUltimateAgent(config)
    
    # إضافة الإصلاحات تلقائياً
    agent.idempotency = DistributedIdempotencyKeys()
    agent.memory_guard = MemoryGuard()
    agent.rollback = MultiVersionRollback()
    agent.health_server = HealthCheckServer()
    agent.shutdown = GracefulShutdown()
    agent.crash_reporter = CrashReporter()
    agent.crash_reporter.install()
    
    # بدء المكونات
    await agent.memory_guard.start()
    await agent.health_server.start()
    
    # بدء الوكيل الأصلي
    await agent.start()
    
    # انتظار إشارة الإيقاف
    await agent.shutdown.wait()

if __name__ == "__main__":
    try:
        asyncio.run(main())
    except KeyboardInterrupt:
        print("\nتم إيقاف البوت")
