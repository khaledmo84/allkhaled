# =============================================================================
# Dockerfile واحد كامل لمشروع Al-Khaled Ultimate
# =============================================================================

FROM python:3.11-slim-bookworm

ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1 \
    TZ=UTC

WORKDIR /app

# تثبيت dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
    gcc g++ make curl git \
    libffi-dev libssl-dev libgmp-dev \
    && apt-get clean && rm -rf /var/lib/apt/lists/*

# نسخ وتثبيت المتطلبات
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# نسخ الكود
COPY . .

# إنشاء المجلدات
RUN mkdir -p logs backups plugins data .alkhaled_persistent

# المنفذ
EXPOSE 8080

# التشغيل المباشر (بدون entrypoint.sh)
CMD ["python", "-u", "unified.py"]
