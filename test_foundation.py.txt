import asyncio
import sys
import os
import tempfile
import unittest
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))

from foundation import (
    SmartHTTPClient, MultiVersionRollback, DistributedIdempotencyKeys,
    SyntaxValidator, FoundationManager
)
from manager import AIHordeClient, PluginManager


class TestFoundationStandalone(unittest.IsolatedAsyncioTestCase):
    """«Œ »«—«  „” Þ·… ·Ð foundation.py"""

    async def test_smart_http(self):
        async with SmartHTTPClient() as c:
            resp = await c.get("https://httpbin.org/get")
            self.assertEqual(resp.status, 200)

    async def test_idempotency(self):
        idem = DistributedIdempotencyKeys(use_upstash=False)
        key = "test_x"
        self.assertFalse(await idem.is_processed(key))
        await idem.mark_processed(key)
        self.assertTrue(await idem.is_processed(key))
        await idem.close()

    async def test_rollback(self):
        rb = MultiVersionRollback()
        with tempfile.NamedTemporaryFile(mode='w', delete=False) as f:
            f.write("original")
            path = f.name
        sid = await rb.create_snapshot(path)
        with open(path, 'w') as f:
            f.write("changed")
        await rb.rollback(sid, path)
        with open(path) as f:
            self.assertEqual(f.read(), "original")
        os.unlink(path)

    async def test_syntax(self):
        errors = SyntaxValidator.validate_all(".")
        self.assertIsInstance(errors, dict)

    async def test_manager(self):
        m = FoundationManager()
        await m.start()
        await m.stop()


class TestManagerStandalone(unittest.IsolatedAsyncioTestCase):
    """«Œ »«—«  „” Þ·… ·Ð manager.py"""

    async def test_ai_horde(self):
        horde = AIHordeClient()
        if await horde.is_available():
            ok, msg = await horde.test_connection()
            self.assertTrue(ok, msg)
        await horde.close()

    async def test_plugin_manager(self):
        pm = PluginManager()
        tools = pm.discover()
        self.assertIsInstance(tools, dict)


def run():
    print("\n" + "=" * 60)
    print(" ‘€Ì· «Œ »«—«  FOUNDATION + MANAGER («·‰”Œ… «·„” Þ·…)")
    print("=" * 60)

    loader = unittest.TestLoader()
    suite = unittest.TestSuite()
    suite.addTests(loader.loadTestsFromTestCase(TestFoundationStandalone))
    suite.addTests(loader.loadTestsFromTestCase(TestManagerStandalone))

    runner = unittest.TextTestRunner(verbosity=2)
    result = runner.run(suite)

    print("\n?  „ »‰Ã«Õ" if result.wasSuccessful() else "? ð‘·")
    return result.wasSuccessful()


if __name__ == "__main__":
    success = run()
    sys.exit(0 if success else 1)
